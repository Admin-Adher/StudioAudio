# Dépendances du véritable bundle macOS natif (WKWebView + moteur local).
#
# Ce fichier est volontairement indépendant de requirements.txt : Torch 2.8
# ne publie plus de roue macOS Intel, alors que la version 2.2.2 couvre encore
# les Mac x86_64. Chaque build GitHub s'exécute sur l'architecture native visée.

faster-whisper==1.2.1
diarize==0.1.2
gradio==6.20.0
python-docx==1.2.0

# Assistant local Apple Silicon. Qwen3 requiert mlx-lm >= 0.25.2 ; cette
# combinaison reste compatible avec macOS 13, contrairement aux roues MLX
# récentes qui imposent macOS 14.
mlx==0.25.2; platform_machine == "arm64"
mlx-lm==0.25.3; platform_machine == "arm64"
# Gradio 6.20 requiert huggingface-hub >= 1.2 tandis que Transformers 4.x
# l'interdit. La branche Transformers 5 garde Qwen3/MLX compatible et permet
# de partager un runtime Hugging Face unique avec l'interface.
transformers==5.14.1; platform_machine == "arm64"
huggingface-hub==1.24.0

# Runtimes audio disponibles pour arm64 comme x86_64 et compatibles macOS 12+.
av==13.1.0
onnxruntime==1.19.2

# Apple Silicon : moteur actuel. Mac Intel : dernière combinaison Torch audio
# disposant encore de roues macOS x86_64 pour Python 3.12.
torch==2.8.0; platform_machine == "arm64"
torchaudio==2.8.0; platform_machine == "arm64"
torch==2.2.2; platform_machine == "x86_64"
torchaudio==2.2.2; platform_machine == "x86_64"
numpy<2; platform_machine == "x86_64"

# pywebview utilise WKWebView/Cocoa : il s'agit bien d'une fenêtre
# d'application locale, et non d'un site distant emballé dans Electron.
pywebview>=5.4,<7
pyinstaller>=6.11,<7
